SCHEMA_NAME(schema_id) AS schema_name,
C.name AS column_name

JOIN sys.columns C ON T.OBJECT_ID = C.OBJECT_ID
WHERE C.name LIKE '%<enter column name>%'     --Adjust this line to search on column or table
